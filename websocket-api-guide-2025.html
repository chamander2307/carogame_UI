class CaroWebSocketClient { constructor(token, roomId) { this.token = token; this.roomId = roomId; this.stompClient = null; this.subscriptions = []; this.isConnected = false; } connect() { const socket = new WebSocket(`ws://localhost:8080/ws?token=Bearer ${this.token}`); this.stompClient = Stomp.over(socket); this.stompClient.connect({}, frame => this.onConnected(frame), error => this.onError(error) ); } onConnected(frame) { console.log('Connected:', frame); this.isConnected = true; // Subscribe to all required topics this.subscribeToTopics(); // Join room automatically this.joinRoom(); } subscribeToTopics() { // Room events this.subscriptions.push( this.stompClient.subscribe(`/topic/room/${this.roomId}`, message => this.handleRoomEvent(JSON.parse(message.body)) ) ); // Game moves this.subscriptions.push( this.stompClient.subscribe(`/topic/game/${this.roomId}/move`, message => this.handleMoveEvent(JSON.parse(message.body)) ) ); // Game end events this.subscriptions.push( this.stompClient.subscribe(`/topic/game/${this.roomId}/end`, message => this.handleGameEndEvent(JSON.parse(message.body)) ) ); // Chat messages this.subscriptions.push( this.stompClient.subscribe(`/topic/room/${this.roomId}/chat`, message => this.handleChatEvent(JSON.parse(message.body)) ) ); // Personal notifications this.subscriptions.push( this.stompClient.subscribe('/user/queue/notifications', message => this.handleNotification(JSON.parse(message.body)) ) ); } handleRoomEvent(event) { switch(event.type) { case 'PLAYER_JOINED': this.updatePlayerList(event.data); break; case 'PLAYER_READY': this.updatePlayerReadyState(event.data); break; case 'GAME_STARTED': this.startGame(event.data); break; case 'REMATCH_REQUESTED': this.showRematchAcceptButton(event.data); break; case 'REMATCH_CREATED': this.redirectToNewRoom(event.data.newRoomId); break; } } // Enhanced methods for new features markReady() { if (this.isConnected) { this.stompClient.send(`/app/room/${this.roomId}/ready`, {}, '{}'); } } requestRematch() { if (this.isConnected) { this.stompClient.send(`/app/room/${this.roomId}/rematch/request`, {}, '{}'); } } acceptRematch() { if (this.isConnected) { this.stompClient.send(`/app/room/${this.roomId}/rematch/accept`, {}, '{}'); } } surrender() { if (this.isConnected) { this.stompClient.send(`/app/room/${this.roomId}/surrender`, {}, '{}'); } } // Handle game end events (win, draw, surrender) handleGameEndEvent(event) { console.log('Game ended:', event); break; } this.disableGameBoard(); this.showRematchButton(); } disconnect() { if (this.stompClient && this.isConnected) { // Clean up subscriptions this.subscriptions.forEach(sub => sub.unsubscribe());
